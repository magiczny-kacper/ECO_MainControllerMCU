# Application build
add_executable(${PROJECT_NAME}_app.elf ${APP_SOURCE_FILES})

set(APP_HEX_FILE ${PROJECT_SOURCE_DIR}/out/${PROJECT_NAME}_app.hex)
set(APP_BIN_FILE ${PROJECT_SOURCE_DIR}/out/${PROJECT_NAME}_app.bin)
add_custom_command(
		TARGET ${PROJECT_NAME}_app.elf POST_BUILD
        COMMAND ${CMAKE_OBJCOPY} -Oihex $<TARGET_FILE:${PROJECT_NAME}_app.elf> ${APP_HEX_FILE}
        COMMAND ${CMAKE_OBJCOPY} -Obinary $<TARGET_FILE:${PROJECT_NAME}_app.elf> ${APP_BIN_FILE}
        COMMAND ${CMAKE_SZ} $<
        COMMENT "Building ${APP_HEX_FILE} \nBuilding ${BIN_FILE}")
 set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -T ${CMAKE_SOURCE_DIR}/${MCU_APP_LINKER_SCRIPT} -Wl,-Map=${PROJECT_SOURCE_DIR}/${PROJECT_NAME}.map")
 
# Bootloader build
add_executable(${PROJECT_NAME}_boot.elf ${BOOT_SOURCE_FILES}) 

set(BOOT_HEX_FILE ${PROJECT_SOURCE_DIR}/out/${PROJECT_NAME}_boot.hex)
set(BOOT_BIN_FILE ${PROJECT_SOURCE_DIR}/out/${PROJECT_NAME}_boot.bin)
add_custom_command(TARGET ${PROJECT_NAME}_boot.elf POST_BUILD
        COMMAND ${CMAKE_OBJCOPY} --pad-to=0x8000 --gap-fill=0xFF -Oihex $<TARGET_FILE:${PROJECT_NAME}_boot.elf> ${BOOT_HEX_FILE}
        COMMAND ${CMAKE_OBJCOPY} --pad-to=0x8000 --gap-fill=0xFF -Obinary $<TARGET_FILE:${PROJECT_NAME}_boot.elf> ${BOOT_BIN_FILE}
        COMMAND ${CMAKE_SZ} $<
        COMMENT "Building ${BOOT_HEX_FILE} \nBuilding ${BOOT_BIN_FILE}")

# Concatenate Bootloader and Application
add_custom_command(TARGET ${PROJECT_NAME}_boot.elf POST_BUILD
	COMMAND ${PROJECT_NAME}.bin: ${BOOT_BIN_FILE} ${APP_BIN_FILE} cat $^ > $@
)
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -T ${CMAKE_SOURCE_DIR}/${MCU_BOOT_LINKER_SCRIPT} -Wl,-Map=${PROJECT_SOURCE_DIR}/${PROJECT_NAME}.map")






# Prawie działają
# Bootloader includes
include_directories(BootloaderSrc)

set(CMAKE_EXE_LINKER_FLAGS "${LD_FLAGS} -T ${CMAKE_SOURCE_DIR}/${MCU_APP_LINKER_SCRIPT} -Wl,-Map=${PROJECT_SOURCE_DIR}/${PROJECT_NAME}_app.map")
add_arm_executable(${CMAKE_PROJECT_NAME}_app  ${APP_SOURCE_FILES})

set(CMAKE_EXE_LINKER_FLAGS "${LD_FLAGS} -T ${CMAKE_SOURCE_DIR}/${MCU_BOOT_LINKER_SCRIPT} -Wl,-Map=${PROJECT_SOURCE_DIR}/${PROJECT_NAME}_boot.map")
add_arm_executable(${CMAKE_PROJECT_NAME}_boot  ${BOOT_SOURCE_FILES})
